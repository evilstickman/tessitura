# M1.6: Grid View UI — Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a view-only grid detail page with cell completion/undo interactions, freshness color mapping, fade toggle, and a landing page linking to grids.

**Architecture:** Director/Presentation pattern. Directors own TanStack Query state and mutations; presentation components are pure (props only). The grid renders as a flush-bordered HTML table (spreadsheet style). Refetch-after-mutate pattern — no optimistic updates.

**Tech Stack:** Next.js 16 (App Router), React 19, TypeScript, Tailwind CSS 4, TanStack Query, Vitest + React Testing Library (unit), Playwright (e2e)

**Spec:** `docs/specs/2026-03-18-m1.6-grid-view-ui-design.md`

```tsx
'use client';

import { QueryClient, QueryClientProvider } from '@tanstack/react-query';
import { useState } from 'react';

export function QueryProvider({ children }: { children: React.ReactNode }) {
  const [queryClient] = useState(
    () =>
      new QueryClient({
        defaultOptions: {
          queries: {
            staleTime: 0,
            retry: 3,
          },
        },
      }),
  );

  return (
    <QueryClientProvider client={queryClient}>
      {children}
    </QueryClientProvider>
  );
}
```

Note: `useState` for QueryClient ensures one instance per client-side session, not recreated on every render. `'use client'` is required because QueryClientProvider uses React context.

```tsx
type FreshnessState = 'incomplete' | 'fresh' | 'aging' | 'stale' | 'decayed';

interface PracticeCellProps {
  cellId: string;
  stepNumber: number;
  targetTempoBpm: number;
  freshnessState: FreshnessState;
  lastCompletionDate: string | null;
  onComplete: (cellId: string) => void;
  onUndo: (cellId: string) => void;
}

const CELL_COLORS: Record<FreshnessState, { bg: string; text: string }> = {
  incomplete: { bg: '#1f2937', text: '#9ca3af' },
  fresh:      { bg: '#10b981', text: '#ffffff' },
  aging:      { bg: '#6ee7b7', text: '#064e3b' },
  stale:      { bg: '#a7f3d0', text: '#064e3b' },
  decayed:    { bg: '#1f2937', text: '#9ca3af' },
};

/** Format YYYY-MM-DD → M-DD (strip leading zeros, drop year). String manipulation only. */
function formatDateDisplay(dateStr: string): string {
  const parts = dateStr.split('-');
  const month = String(Number(parts[1]));
  const day = String(Number(parts[2]));
  return `${month}-${day}`;
}

function isCompleted(state: FreshnessState): boolean {
  return state === 'fresh' || state === 'aging' || state === 'stale';
}

export function PracticeCell({
  cellId,
  stepNumber,
  targetTempoBpm,
  freshnessState,
  lastCompletionDate,
  onComplete,
  onUndo,
}: PracticeCellProps) {
  const colors = CELL_COLORS[freshnessState];
  const completed = isCompleted(freshnessState);
  const displayText = completed && lastCompletionDate
    ? formatDateDisplay(lastCompletionDate)
    : String(targetTempoBpm);

  // stepNumber is 0-indexed; display as 1-indexed for aria
  const stepLabel = stepNumber + 1;
  const ariaLabel = completed && lastCompletionDate
    ? `Undo step ${stepLabel}, completed ${formatDateDisplay(lastCompletionDate)}`
    : `Complete step ${stepLabel} at ${targetTempoBpm} BPM`;

  function handleContextMenu(e: React.MouseEvent) {
    e.preventDefault();
    onUndo(cellId);
  }

  return (
    <td
      style={{
        backgroundColor: colors.bg,
        border: '1px solid #374151',
        textAlign: 'center',
        width: '52px',
        fontSize: '11px',
        padding: 0,
      }}
    >
      <button
        type="button"
        onClick={() => onComplete(cellId)}
        onContextMenu={handleContextMenu}
        aria-label={ariaLabel}
        style={{
          width: '100%',
          height: '100%',
          padding: '8px 4px',
          background: 'transparent',
          border: 'none',
          color: colors.text,
          fontSize: 'inherit',
          cursor: 'pointer',
          fontFamily: 'inherit',
        }}
      >
        {displayText}
      </button>
    </td>
  );
}
```

```tsx
import { PracticeCell } from '@/components/presentation/PracticeCell';

interface CellData {
  id: string;
  stepNumber: number;
  targetTempoBpm: number;
  freshnessState: 'incomplete' | 'fresh' | 'aging' | 'stale' | 'decayed';
  lastCompletionDate: string | null;
}

interface PieceData {
  id: string;
  title: string;
  composer: string | null;
}

interface RowData {
  id: string;
  piece: PieceData | null;
  passageLabel: string | null;
  startMeasure: number;
  endMeasure: number;
  targetTempo: number;
  priority: string;
  cells: CellData[];
}

interface GridRowProps {
  row: RowData;
  onComplete: (rowId: string, cellId: string) => void;
  onUndo: (rowId: string, cellId: string) => void;
}

const PRIORITY_COLORS: Record<string, string> = {
  CRITICAL: '#ef4444',
  HIGH: '#fbbf24',
  MEDIUM: '#9ca3af',
  LOW: '#9ca3af',
};

function getRowLabel(piece: PieceData | null, passageLabel: string | null): string {
  if (piece && passageLabel) return `${piece.title} — ${passageLabel}`;
  if (piece) return piece.title;
  if (passageLabel) return passageLabel;
  return 'Untitled';
}

export function GridRow({ row, onComplete, onUndo }: GridRowProps) {
  const label = getRowLabel(row.piece, row.passageLabel);
  const priorityColor = PRIORITY_COLORS[row.priority] ?? '#9ca3af';

  return (
    <tr>
      <td
        style={{
          padding: '8px 10px',
          border: '1px solid #374151',
          backgroundColor: '#1f2937',
          minWidth: '140px',
        }}
      >
        <div style={{ color: '#f9fafb', fontWeight: 600, fontSize: '13px' }}>
          {label}
        </div>
        <div style={{ color: '#6b7280', fontSize: '11px' }}>
          mm. {row.startMeasure}–{row.endMeasure} ·{' '}
          <span style={{ color: priorityColor }}>{row.priority}</span>
        </div>
      </td>
      {row.cells.map((cell) => (
        <PracticeCell
          key={cell.id}
          cellId={cell.id}
          stepNumber={cell.stepNumber}
          targetTempoBpm={cell.targetTempoBpm}
          freshnessState={cell.freshnessState}
          lastCompletionDate={cell.lastCompletionDate}
          onComplete={(cellId) => onComplete(row.id, cellId)}
          onUndo={(cellId) => onUndo(row.id, cellId)}
        />
      ))}
    </tr>
  );
}
```

```tsx
'use client';

import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { GridHeader } from '@/components/presentation/GridHeader';
import { GridTable } from '@/components/presentation/GridTable';

interface GridViewDirectorProps {
  gridId: string;
}

async function fetchGrid(gridId: string) {
  const res = await fetch(`/api/grids/${gridId}`);
  if (res.status === 404) return null;
  if (res.status === 401) {
    window.location.href = '/';
    return null;
  }
  if (!res.ok) throw new Error(`Failed to fetch grid: ${res.status}`);
  return res.json();
}

export function GridViewDirector({ gridId }: GridViewDirectorProps) {
  const queryClient = useQueryClient();
  const queryKey = ['grid', gridId];

  const { data: grid, isLoading, error } = useQuery({
    queryKey,
    queryFn: () => fetchGrid(gridId),
  });

  const completeMutation = useMutation({
    mutationFn: async ({ rowId, cellId }: { rowId: string; cellId: string }) => {
      const res = await fetch(`/api/grids/${gridId}/rows/${rowId}/cells/${cellId}/complete`, {
        method: 'POST',
      });
      if (!res.ok && res.status !== 409 && res.status !== 404) {
        throw new Error(`Complete failed: ${res.status}`);
      }
    },
    onSettled: () => queryClient.invalidateQueries({ queryKey }),
  });

  const undoMutation = useMutation({
    mutationFn: async ({ rowId, cellId }: { rowId: string; cellId: string }) => {
      const res = await fetch(`/api/grids/${gridId}/rows/${rowId}/cells/${cellId}/undo`, {
        method: 'POST',
      });
      if (!res.ok && res.status !== 409 && res.status !== 404) {
        throw new Error(`Undo failed: ${res.status}`);
      }
    },
    onSettled: () => queryClient.invalidateQueries({ queryKey }),
  });

  const fadeMutation = useMutation({
    mutationFn: async (fadeEnabled: boolean) => {
      const res = await fetch(`/api/grids/${gridId}/fade`, {
        method: 'PUT',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ fadeEnabled }),
      });
      if (!res.ok) throw new Error(`Fade toggle failed: ${res.status}`);
    },
    onSettled: () => queryClient.invalidateQueries({ queryKey }),
  });

  function handleComplete(rowId: string, cellId: string) {
    completeMutation.mutate({ rowId, cellId });
  }

  function handleUndo(rowId: string, cellId: string) {
    undoMutation.mutate({ rowId, cellId });
  }

  function handleToggleFade() {
    if (grid) fadeMutation.mutate(!grid.fadeEnabled);
  }

  if (isLoading) return <div style={{ padding: '20px', color: '#9ca3af' }}>Loading...</div>;
  if (error || !grid) return <div style={{ padding: '20px', color: '#9ca3af' }}>Grid not found</div>;

  // Map API response rows to the shape presentation components expect
  const rows = grid.rows.map((row: Record<string, unknown>) => ({
    id: row.id,
    piece: row.piece,
    passageLabel: row.passageLabel,
    startMeasure: row.startMeasure,
    endMeasure: row.endMeasure,
    targetTempo: row.targetTempo,
    priority: row.priority,
    cells: (row.cells as Record<string, unknown>[]).map((cell) => ({
      id: cell.id,
      stepNumber: cell.stepNumber,
      targetTempoBpm: cell.targetTempoBpm,
      freshnessState: cell.freshnessState,
      lastCompletionDate: cell.lastCompletionDate,
    })),
  }));

  return (
    <div>
      <GridHeader
        name={grid.name}
        notes={grid.notes}
        fadeEnabled={grid.fadeEnabled}
        completionPercentage={grid.completionPercentage}
        freshnessSummary={grid.freshnessSummary}
        onToggleFade={handleToggleFade}
      />
      <GridTable
        rows={rows}
        onComplete={handleComplete}
        onUndo={handleUndo}
      />
    </div>
  );
}
```
