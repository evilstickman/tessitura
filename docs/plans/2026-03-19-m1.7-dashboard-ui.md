# M1.7: Dashboard UI — Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a 4-quadrant dashboard at `/` with alerts, grid list with stats, statistics, and practice focus. Includes enriched grid list API (`?detail=true`), grid creation form, and a `/grids` page for the full grid list.

**Architecture:** Backend enrichment adds `?detail=true` to `GET /api/grids` returning freshness stats per grid and per row. Frontend uses Director/Presentation pattern — single DashboardDirector fetches enriched data, derives alerts/stats/focus, passes to 4 pane components. Grid creation via existing `POST /api/grids`.

**Tech Stack:** Next.js 16 (App Router), React 19, TypeScript, TanStack Query, Vitest + React Testing Library, Playwright

```ts
              completions: {

              },

          },

    },
  });
```

```ts
/**
 * Formats grids with freshness summaries but without cell-level detail.
 * Used by GET /api/grids?detail=true for the dashboard.
 *
 * For each grid: computes completionPercentage + freshnessSummary.
 * For each row: computes row-level completionPercentage + freshnessSummary,
 * strips cells/sortOrder/targetTempo/steps/timestamps.
 */
export function formatGridSummaryList(grids: GridDetailRecord[], now: Date) {
  return grids.map((grid) => {
    // Use formatRow to compute freshness, then strip cell data
    const fullRows = grid.practiceRows.map((row) => formatRow(row, grid.fadeEnabled, now));

    const allCellStates: CellWithEffectiveState[] = fullRows.flatMap((row) =>
      row.cells.map((cell) => ({ effectiveState: cell.freshnessState }))
    );

    const rowSummaries = fullRows.map((row) => ({
      id: row.id,
      piece: row.piece,
      passageLabel: row.passageLabel,
      startMeasure: row.startMeasure,
      endMeasure: row.endMeasure,
      priority: row.priority,
      completionPercentage: row.completionPercentage,
      freshnessSummary: row.freshnessSummary,
    }));

    return {
      ...formatGrid(grid),
      completionPercentage: calculateCompletionPercentage(allCellStates, grid.fadeEnabled),
      freshnessSummary: calculateFreshnessSummary(allCellStates),
      rows: rowSummaries,
    };
  });
}
```

```ts
export async function listGrids(request: NextRequest) {
  try {
    const userId = await getCurrentUserId();
    const url = new URL(request.url);
    const detail = url.searchParams.get('detail') === 'true';

    if (detail) {
      const now = new Date();
      const grids = await listGridsWithDetailModel(userId);
      return NextResponse.json(formatGridSummaryList(grids, now));
    }

    const grids = await listGridsModel(userId);
    return NextResponse.json(formatGridList(grids));
```

```tsx
import Link from 'next/link';

interface AlertItem {
  id: string;
  gridName: string;
  count: number;
  type: 'decayed' | 'stale';
  href: string;
}

interface AlertsPaneProps {
  alerts: AlertItem[];
  hasGrids?: boolean;
}

export function AlertsPane({ alerts, hasGrids = true }: AlertsPaneProps) {
  const displayAlerts = alerts.slice(0, 5);
  const hasMore = alerts.length > 5;

  return (
    <div>
      <div style={{ color: '#6b7280', fontSize: '11px', textTransform: 'uppercase', letterSpacing: '0.05em', marginBottom: '12px' }}>
        Alerts
      </div>

      {!hasGrids && alerts.length === 0 ? (
        <p style={{ color: '#9ca3af', fontSize: '13px' }}>
          Create your first practice grid to get started
        </p>
      ) : (
        <>
          <div style={{ color: '#f9fafb', fontSize: '15px', fontWeight: 500, marginBottom: '14px' }}>
            Welcome back, Dev User
          </div>
          <div style={{ display: 'flex', flexDirection: 'column', gap: '8px' }}>
            {displayAlerts.map((alert) => (
              <Link
                key={alert.id}
                href={alert.href}
                style={{
                  display: 'flex',
                  alignItems: 'center',
                  gap: '8px',
                  padding: '8px 10px',
                  background: '#111827',
                  borderRadius: '4px',
                  textDecoration: 'none',
                  borderLeft: `3px solid ${alert.type === 'decayed' ? '#ef4444' : '#fbbf24'}`,
                }}
              >
                <div style={{ flex: 1 }}>
                  <div style={{ color: '#f9fafb', fontSize: '12px' }}>
                    {alert.type === 'decayed' ? `Resume ${alert.gridName}` : alert.gridName}
                  </div>
                  <div style={{ color: '#9ca3af', fontSize: '11px' }}>
                    {alert.type === 'decayed'
                      ? `${alert.count} cells need re-practice`
                      : `${alert.count} cells losing freshness`}
                  </div>
                </div>
                <span style={{ color: '#6b7280', fontSize: '11px' }}>→</span>
              </Link>
            ))}
          </div>
          {hasMore && (
            <div style={{ textAlign: 'right', marginTop: '8px' }}>
              <Link href="/grids" style={{ color: '#6b7280', fontSize: '11px', textDecoration: 'none' }}>
                See all →
              </Link>
            </div>
          )}
        </>
      )}
    </div>
  );
}
```

```tsx
'use client';

import { useState } from 'react';

interface GridCreateFormProps {
  onSubmit: (name: string, notes: string) => void;
  onCancel: () => void;
  error?: string | null;
}

export function GridCreateForm({ onSubmit, onCancel, error }: GridCreateFormProps) {
  const [name, setName] = useState('');
  const [notes, setNotes] = useState('');

  function handleSubmit(e: React.FormEvent) {
    e.preventDefault();
    if (name.trim().length === 0) return;
    onSubmit(name.trim(), notes.trim());
  }

  const inputStyle = {
    width: '100%',
    padding: '6px 8px',
    background: '#111827',
    color: '#f9fafb',
    border: '1px solid #374151',
    borderRadius: '4px',
    fontSize: '12px',
    marginBottom: '8px',
  };

  return (
    <form onSubmit={handleSubmit} style={{ marginBottom: '12px' }}>
      <input
        type="text"
        placeholder="Grid name"
        value={name}
        onChange={(e) => setName(e.target.value)}
        style={inputStyle}
      />
      <input
        type="text"
        placeholder="Notes (optional)"
        value={notes}
        onChange={(e) => setNotes(e.target.value)}
        style={inputStyle}
      />
      <div style={{ display: 'flex', gap: '8px' }}>
        <button
          type="submit"
          style={{ background: '#10b981', color: 'white', border: 'none', padding: '4px 10px', borderRadius: '4px', fontSize: '11px', cursor: 'pointer' }}
        >
          Create
        </button>
        <button
          type="button"
          onClick={onCancel}
          style={{ background: '#374151', color: '#9ca3af', border: 'none', padding: '4px 10px', borderRadius: '4px', fontSize: '11px', cursor: 'pointer' }}
        >
          Cancel
        </button>
      </div>
      {error && <div style={{ color: '#ef4444', fontSize: '11px', marginTop: '6px' }}>{error}</div>}
    </form>
  );
}
```

```tsx
import Link from 'next/link';
import { GridCreateForm } from '@/components/presentation/GridCreateForm';

interface GridCardData {
  id: string;
  name: string;
  completionPercentage: number;
  updatedAt: string;
  attentionCount: number;
}

interface MyGridsPaneProps {
  grids: GridCardData[];
  onNewGrid: () => void;
  showForm: boolean;
  onSubmitGrid: (name: string, notes: string) => void;
  onCancelForm: () => void;
  formError?: string | null;
}

function formatUpdatedDate(isoDate: string): string {
  const parts = isoDate.slice(0, 10).split('-');
  const month = String(Number(parts[1]));
  const day = String(Number(parts[2]));
  return `Updated ${month}/${day}`;
}

export function MyGridsPane({ grids, onNewGrid, showForm, onSubmitGrid, onCancelForm, formError }: MyGridsPaneProps) {
  const displayGrids = grids.slice(0, 5);
  const hasMore = grids.length > 5;

  return (
    <div>
      <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', marginBottom: '12px' }}>
        <div style={{ color: '#6b7280', fontSize: '11px', textTransform: 'uppercase', letterSpacing: '0.05em' }}>
          My Grids
        </div>
        <button
          type="button"
          onClick={onNewGrid}
          style={{ background: '#10b981', color: 'white', border: 'none', padding: '4px 10px', borderRadius: '4px', fontSize: '11px', cursor: 'pointer' }}
        >
          + New Grid
        </button>
      </div>

      {showForm && (
        <GridCreateForm onSubmit={onSubmitGrid} onCancel={onCancelForm} error={formError} />
      )}

      {grids.length === 0 && !showForm ? (
        <p style={{ color: '#9ca3af', fontSize: '13px' }}>No practice grids yet.</p>
      ) : (
        <div style={{ display: 'flex', flexDirection: 'column', gap: '6px' }}>
          {displayGrids.map((grid) => (
            <Link
              key={grid.id}
              href={`/grids/${grid.id}`}
              style={{
                display: 'flex',
                alignItems: 'center',
                padding: '8px 10px',
                background: '#111827',
                borderRadius: '4px',
                textDecoration: 'none',
                gap: '12px',
              }}
            >
              <div style={{ flex: 1 }}>
                <div style={{ color: '#f9fafb', fontSize: '12px' }}>{grid.name}</div>
                <div style={{ color: '#6b7280', fontSize: '11px' }}>
                  {formatUpdatedDate(grid.updatedAt)}
                  {grid.attentionCount > 0 && (
                    <span style={{ color: '#fbbf24', marginLeft: '8px' }}>
                      {grid.attentionCount} cells need attention
                    </span>
                  )}
                </div>
              </div>
              <div style={{ width: '60px', height: '4px', background: '#374151', borderRadius: '2px', overflow: 'hidden' }}>
                <div style={{ width: `${Math.round(grid.completionPercentage)}%`, height: '100%', background: '#10b981' }} />

              <span style={{ color: '#9ca3af', fontSize: '11px', minWidth: '30px', textAlign: 'right' }}>
                {Math.round(grid.completionPercentage)}%

            </Link>

        </div>

        <div style={{ textAlign: 'right', marginTop: '8px' }}>
          <Link href="/grids" style={{ color: '#6b7280', fontSize: '11px', textDecoration: 'none' }}>
            See all →
          </Link>
        </div>
      )}
    </div>
  );
}
```

```tsx
'use client';

import { useState } from 'react';
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { DashboardLayout } from '@/components/presentation/DashboardLayout';
import { AlertsPane } from '@/components/presentation/AlertsPane';
import { MyGridsPane } from '@/components/presentation/MyGridsPane';
import { StatisticsPane } from '@/components/presentation/StatisticsPane';
import { PracticeFocusPane } from '@/components/presentation/PracticeFocusPane';

interface FreshnessSummary {
  fresh: number;
  aging: number;
  stale: number;
  decayed: number;
  incomplete: number;
}

interface ApiRowSummary {
  id: string;
  piece: { title: string; composer: string | null } | null;
  passageLabel: string | null;
  startMeasure: number;
  endMeasure: number;
  priority: string;
  completionPercentage: number;
  freshnessSummary: FreshnessSummary;
}

interface ApiGridSummary {
  id: string;
  name: string;
  notes: string | null;
  fadeEnabled: boolean;
  completionPercentage: number;
  freshnessSummary: FreshnessSummary;
  updatedAt: string;
  createdAt: string;
  rows: ApiRowSummary[];
}

class AuthError extends Error {
  constructor() { super('Authentication required'); this.name = 'AuthError'; }
}

async function fetchDashboardGrids(): Promise<ApiGridSummary[]> {
  const res = await fetch('/api/grids?detail=true');
  if (res.status === 401) throw new AuthError();
  if (!res.ok) throw new Error(`Failed to fetch: ${res.status}`);
  return res.json();
}

// ─── Data derivation helpers ─────────────────────────────────────────────────

function getRowLabel(piece: { title: string } | null, passageLabel: string | null): string {
  if (piece && passageLabel) return `${piece.title} — ${passageLabel}`;
  if (piece) return piece.title;
  if (passageLabel) return passageLabel;
  return 'Untitled';
}

const PRIORITY_ORDER: Record<string, number> = { CRITICAL: 0, HIGH: 1, MEDIUM: 2, LOW: 3 };

function deriveAlerts(grids: ApiGridSummary[]) {
  const alerts: { id: string; gridName: string; count: number; type: 'decayed' | 'stale'; href: string }[] = [];

  for (const grid of grids) {
    if (grid.freshnessSummary.decayed > 0) {
      alerts.push({ id: `${grid.id}-decayed`, gridName: grid.name, count: grid.freshnessSummary.decayed, type: 'decayed', href: `/grids/${grid.id}` });
    }
    if (grid.freshnessSummary.stale > 0) {
      alerts.push({ id: `${grid.id}-stale`, gridName: grid.name, count: grid.freshnessSummary.stale, type: 'stale', href: `/grids/${grid.id}` });
    }
  }

  // Sort: decayed first, then stale. Within type, by count descending.
  alerts.sort((a, b) => {
    if (a.type !== b.type) return a.type === 'decayed' ? -1 : 1;
    return b.count - a.count;
  });

  return alerts;
}

function deriveStats(grids: ApiGridSummary[]) {
  if (grids.length === 0) return { avgCompletion: 0, cellsCompleted: 0 };

  const avgCompletion = grids.reduce((sum, g) => sum + g.completionPercentage, 0) / grids.length;
  const cellsCompleted = grids.reduce((sum, g) => {
    const s = g.freshnessSummary;
    return sum + s.fresh + s.aging + s.stale + s.decayed;
  }, 0);

  return { avgCompletion, cellsCompleted };
}

function derivePracticeFocus(grids: ApiGridSummary[]) {
  const rows: { rowId: string; label: string; gridName: string; gridId: string; priority: string; needsPracticeCount: number }[] = [];

  for (const grid of grids) {
    for (const row of grid.rows) {
      const count = row.freshnessSummary.stale + row.freshnessSummary.decayed;
      rows.push({
        rowId: row.id,
        label: getRowLabel(row.piece, row.passageLabel),
        gridName: grid.name,
        gridId: grid.id,
        priority: row.priority,
        needsPracticeCount: count,
      });
    }
  }

  const needsPractice = rows.filter((r) => r.needsPracticeCount > 0);
  const allFresh = needsPractice.length === 0;

  const toRank = allFresh ? rows : needsPractice;
  toRank.sort((a, b) => {
    const pa = PRIORITY_ORDER[a.priority] ?? 3;
    const pb = PRIORITY_ORDER[b.priority] ?? 3;
    if (pa !== pb) return pa - pb;
    return b.needsPracticeCount - a.needsPracticeCount;
  });

  return toRank.slice(0, 5).map((r) => ({
    rowId: r.rowId,
    label: r.label,
    gridName: r.gridName,
    priority: r.priority,
    needsPracticeCount: r.needsPracticeCount,
    href: `/grids/${r.gridId}`,
    allFresh,
  }));
}

// ─── Component ───────────────────────────────────────────────────────────────

export function DashboardDirector() {
  const queryClient = useQueryClient();
  const [showCreateForm, setShowCreateForm] = useState(false);
  const [formError, setFormError] = useState<string | null>(null);

  const { data: grids, isLoading, error } = useQuery({
    queryKey: ['grids'],
    queryFn: fetchDashboardGrids,
  });

  const createMutation = useMutation({
    mutationFn: async ({ name, notes }: { name: string; notes: string }) => {
      const res = await fetch('/api/grids', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ name, notes: notes || null }),
      });
      if (!res.ok) {
        const body = await res.json().catch(() => null);
        throw new Error(body?.error?.message ?? 'Failed to create grid');
      }
      return res.json();
    },
    onSuccess: () => {
      setShowCreateForm(false);
      setFormError(null);
      queryClient.invalidateQueries({ queryKey: ['grids'] });
    },
    onError: (err: Error) => {
      setFormError(err.message);
    },
  });

  if (isLoading) return <div style={{ padding: '40px', color: '#9ca3af', textAlign: 'center' }}>Loading...</div>;
  if (error instanceof AuthError) return <div style={{ padding: '40px', color: '#9ca3af', textAlign: 'center' }}>Authentication required</div>;
  if (error) return <div style={{ padding: '40px', color: '#9ca3af', textAlign: 'center' }}>Failed to load dashboard.</div>;

  const gridData = grids ?? [];
  const alerts = deriveAlerts(gridData);
  const { avgCompletion, cellsCompleted } = deriveStats(gridData);
  const suggestions = derivePracticeFocus(gridData);
  const hasGrids = gridData.length > 0;

  const gridCards = gridData.map((g) => ({
    id: g.id,
    name: g.name,
    completionPercentage: g.completionPercentage,
    updatedAt: g.updatedAt,
    attentionCount: g.freshnessSummary.stale + g.freshnessSummary.decayed,
  }));

  return (
    <DashboardLayout
      alerts={<AlertsPane alerts={alerts} hasGrids={hasGrids} />}
      grids={
        <MyGridsPane
          grids={gridCards}
          onNewGrid={() => { setShowCreateForm(true); setFormError(null); }}
          showForm={showCreateForm}
          onSubmitGrid={(name, notes) => createMutation.mutate({ name, notes })}
          onCancelForm={() => { setShowCreateForm(false); setFormError(null); }}
          formError={formError}
        />
      }
      stats={<StatisticsPane avgCompletion={avgCompletion} cellsCompleted={cellsCompleted} hasGrids={hasGrids} />}
      focus={<PracticeFocusPane suggestions={suggestions} />}
    />
  );
}
```
