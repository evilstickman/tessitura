```typescript
type ArgsWithWhere = {
  where?: Record<string, unknown>;
  [key: string]: unknown;
};

export function addSoftDeleteFilter<T extends ArgsWithWhere>(args: T): T & { where: Record<string, unknown> } {
  const where = args.where ?? {};

  // Don't overwrite if caller explicitly set deletedAt
  if (!('deletedAt' in where)) {
    where.deletedAt = null;
  }

  return { ...args, where };
}
```

**Step 5: Update db.ts to use the extension**

Update `src/lib/db.ts` to apply soft-delete middleware. Since Prisma 7 uses client extensions, we wrap the client:

```typescript
import { PrismaPg } from '@prisma/adapter-pg';
import { PrismaClient } from '@/generated/prisma/client';
import { addSoftDeleteFilter } from '@/lib/prisma-soft-delete';

const globalForPrisma = globalThis as unknown as {
  prisma: ReturnType<typeof createPrismaClient> | undefined;
};

function createPrismaClient() {
  const adapter = new PrismaPg({ connectionString: process.env.DATABASE_URL });
  const baseClient = new PrismaClient({ adapter });

  return baseClient.$extends({
    query: {
      $allModels: {
        async findMany({ args, query }) {
          return query(addSoftDeleteFilter(args));
        },
        async findFirst({ args, query }) {
          return query(addSoftDeleteFilter(args));
        },
        async findUnique({ args, query }) {
          return query(addSoftDeleteFilter(args));
        },
        async count({ args, query }) {
          return query(addSoftDeleteFilter(args));
        },
        async delete({ model, args, query }) {
          // Convert hard delete to soft delete
          const { where } = args;
          return (baseClient[model as keyof typeof baseClient] as any).update({
            where,
            data: { deletedAt: new Date() },
          });
        },
        async deleteMany({ model, args, query }) {
          // Convert hard deleteMany to soft deleteMany
          return (baseClient[model as keyof typeof baseClient] as any).updateMany({
            ...args,
            data: { deletedAt: new Date() },
          });
        },
      },
    },
  });
}

export const prisma =
  globalForPrisma.prisma ?? createPrismaClient();

if (process.env.NODE_ENV !== 'production') {
  globalForPrisma.prisma = prisma;
}

export type ExtendedPrismaClient = typeof prisma;
```

import { describe, it, expect } from 'vitest';
import { getTestPrisma } from '../../helpers/db';
import { prisma as appPrisma } from '@/lib/db';

describe('Soft delete behavior', () => {
  const rawPrisma = getTestPrisma();

  async function createTestGrid() {
    const user = await rawPrisma.user.create({
      data: { email: 'softdel@example.com', passwordHash: 'hash', displayName: 'Soft' },
    });
    const grid = await rawPrisma.practiceGrid.create({
      data: { userId: user.id, name: 'Soft Delete Grid' },
    });
    return { user, grid };
  }

  it('soft-deletes a record by setting deletedAt', async () => {
    const { grid } = await createTestGrid();

    // Soft delete via app prisma
    await appPrisma.practiceGrid.delete({ where: { id: grid.id } });

    // Raw query should still find it with deletedAt set
    const deleted = await rawPrisma.practiceGrid.findUnique({
      where: { id: grid.id },
    });

    expect(deleted).not.toBeNull();
    expect(deleted?.deletedAt).not.toBeNull();
  });

  it('filters soft-deleted records from findMany', async () => {
    const { user, grid } = await createTestGrid();

    // Soft delete
    await rawPrisma.practiceGrid.update({
      where: { id: grid.id },
      data: { deletedAt: new Date() },
    });

    // App prisma should not find it
    const grids = await appPrisma.practiceGrid.findMany({
      where: { userId: user.id },
    });

    expect(grids).toHaveLength(0);
  });

  it('filters soft-deleted records from findFirst', async () => {
    const { user, grid } = await createTestGrid();

    await rawPrisma.practiceGrid.update({
      where: { id: grid.id },
      data: { deletedAt: new Date() },
    });

    const found = await appPrisma.practiceGrid.findFirst({
      where: { userId: user.id },
    });
