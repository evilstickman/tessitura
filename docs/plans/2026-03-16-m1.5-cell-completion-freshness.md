```typescript
/**
 * Pure freshness calculation functions.
 * No database access — every function accepts a `now` parameter for testability.
 *
 * Design doc: docs/specs/2026-03-16-m1.5-cell-completion-freshness-design.md
 */

export type FreshnessState = 'incomplete' | 'fresh' | 'aging' | 'stale' | 'decayed';

const MS_PER_DAY = 86_400_000;
const MAX_INTERVAL = 30;

/**
 * Calculates the raw freshness state of a cell based on time since last completion.
 *
 * - No completions → incomplete
 * - daysSince <= interval * 0.5 → fresh
 * - daysSince <= interval → aging
 * - daysSince <= interval * 2 → stale
 * - else → decayed
 */
export function calculateFreshnessState(
  lastCompletionDate: Date | null,
  intervalDays: number,
  now: Date,
): FreshnessState {
  if (lastCompletionDate === null) return 'incomplete';

  const daysSince = Math.floor((now.getTime() - lastCompletionDate.getTime()) / MS_PER_DAY);

  if (daysSince <= intervalDays * 0.5) return 'fresh';
  if (daysSince <= intervalDays) return 'aging';
  if (daysSince <= intervalDays * 2) return 'stale';
  return 'decayed';
}

/**
 * Calculates the new interval after a completion.
 * Uses RAW freshness state (not effective/shielded state).
 *
 * - incomplete → 1 (first completion)
 * - fresh/aging → min(current * 2, 30)
 * - stale/decayed → 1 (reset)
 */
export function calculateNewInterval(
  currentInterval: number,
  rawState: FreshnessState,
): number {
  switch (rawState) {
    case 'incomplete':
      return 1;
    case 'fresh':
    case 'aging':
      return Math.min(currentInterval * 2, MAX_INTERVAL);
    case 'stale':
    case 'decayed':
      return 1;
  }
}
```

```typescript
/**
 * Minimal cell shape needed for shielding calculation.
 * Cells must be ordered by targetTempoPercentage ascending (array position 0 = lowest).
 * The array position IS the cell's position in the shielding chain — no separate index field.
 */
export interface CellWithState {
  rawState: FreshnessState;
  hasCompletions: boolean;
}

/**
 * Determines if a cell is shielded by a higher-tempo completed cell.
 *
 * Shielding rules:
 * - Fade disabled → no shielding
 * - Incomplete cells (no non-deleted completions) → never shielded
 * - Walk upward through cells by index, skipping incomplete cells
 * - If the next higher completed cell's raw state is NOT decayed → shielded
 * - If it IS decayed → shield has fallen
 * - If no higher completed cell exists → unshielded (this cell fades first)
 */
export function isShielded(
  cellIndex: number,
  cells: CellWithState[],
  fadeEnabled: boolean,
): boolean {
  if (!fadeEnabled) return false;

  const cell = cells[cellIndex];
  if (!cell.hasCompletions) return false;

  // Find next higher completed cell
  for (let i = cellIndex + 1; i < cells.length; i++) {
    if (cells[i].hasCompletions) {
      return cells[i].rawState !== 'decayed';
    }
  }

  // No higher completed cell — this is the highest, fades first
  return false;
}

/**
 * Calculates the effective freshness state (what the user sees).
 *
 * - Incomplete is always incomplete
 * - Fade disabled → all completed cells appear fresh (permanent green)
 * - Shielded → appears fresh
 * - Otherwise → raw state
 */
export function effectiveFreshnessState(
  rawState: FreshnessState,
  isShieldedFlag: boolean,
  fadeEnabled: boolean,
): FreshnessState {
  if (rawState === 'incomplete') return 'incomplete';
  if (!fadeEnabled) return 'fresh';
  if (isShieldedFlag) return 'fresh';
  return rawState;
}
```

```typescript
export interface CellWithEffectiveState {
  effectiveState: FreshnessState;
}

/**
 * Calculates the completion percentage for a set of cells.
 *
 * - Fade ON: fresh + aging + stale count (decayed + incomplete excluded)
 * - Fade OFF: all non-incomplete count
 * - Zero cells → 0
 */
export function calculateCompletionPercentage(
  cells: CellWithEffectiveState[],
  fadeEnabled: boolean,
): number {
  if (cells.length === 0) return 0;

  const countCompleted = fadeEnabled
    ? cells.filter((c) => c.effectiveState === 'fresh' || c.effectiveState === 'aging' || c.effectiveState === 'stale').length
    : cells.filter((c) => c.effectiveState !== 'incomplete').length;

  return (countCompleted / cells.length) * 100;
}

/**
 * Counts cells by their effective freshness state.
 */
export function calculateFreshnessSummary(
  cells: CellWithEffectiveState[],
): { fresh: number; aging: number; stale: number; decayed: number; incomplete: number } {
  const summary = { fresh: 0, aging: 0, stale: 0, decayed: 0, incomplete: 0 };
  for (const cell of cells) {
    summary[cell.effectiveState]++;
  }
  return summary;
}
```

```typescript
  // Prisma returns by stepNumber:asc which SHOULD match targetTempoPercentage order
  // (cells are created in ascending tempo order), but we sort explicitly to
  // guarantee correctness regardless of creation order.

    .sort((a, b) => a.targetTempoPercentage - b.targetTempoPercentage);

  // Step 2: Compute raw states for all cells
  const cellStates: CellWithState[] = sortedCells.map((cell, index) => {
    const lastCompletion = cell.completions.length > 0
      ? cell.completions[cell.completions.length - 1].completionDate
      : null;
    const rawState = calculateFreshnessState(lastCompletion, cell.freshnessIntervalDays, now);
    return {
      index,
      rawState,
      hasCompletions: cell.completions.length > 0,
    };

  const effectiveStates = cellStates.map((_, index) => {
    const shielded = isShielded(index, cellStates, fadeEnabled);
    return effectiveFreshnessState(cellStates[index].rawState, shielded, fadeEnabled);
```

```typescript
import { prisma } from '@/lib/db';
import { ConflictError } from '@/lib/errors';
import { calculateFreshnessState, calculateNewInterval } from '@/models/freshness';
import type { PrismaClient } from '@/generated/prisma/client';

type TransactionClient = Parameters<Parameters<PrismaClient['$transaction']>[0]>[0];

/**
 * Finds a cell by ID, verifying the full ownership chain:
 * grid (owned by user, not deleted) → row (belongs to grid, not deleted) → cell (not deleted).
 * All checks are query-driven.
 *
 * Accepts a transaction client so it can run inside a transaction (required for ACID).
 */
async function findOwnedCell(
  tx: TransactionClient,
  gridId: string,
  rowId: string,
  cellId: string,
  userId: string,
) {
  const grid = await tx.practiceGrid.findFirst({
    where: { id: gridId, userId, deletedAt: null },
  });
  if (!grid) return null;

  const row = await tx.practiceRow.findFirst({
    where: { id: rowId, practiceGridId: gridId, deletedAt: null },
  });
  if (!row) return null;

  const cell = await tx.practiceCell.findFirst({
    where: { id: cellId, practiceRowId: rowId, deletedAt: null },
    include: {
      completions: {
        where: { deletedAt: null },
        orderBy: { completionDate: 'asc' },
      },
    },
  });

  return cell;
}

/**
 * Include clause for cell action returns — includes siblings for shielding computation.
 * Completions ordered by completionDate asc — last element is most recent (intentional).
 */
const CELL_RETURN_INCLUDE = {
  completions: {
    where: { deletedAt: null },
    orderBy: { completionDate: 'asc' as const },
  },
  practiceRow: {
    select: {
      targetTempo: true,
      practiceGrid: { select: { fadeEnabled: true } },
      practiceCells: {
        where: { deletedAt: null },
        orderBy: { stepNumber: 'asc' as const },
        include: {
          completions: {
            where: { deletedAt: null },
            orderBy: { completionDate: 'asc' as const },
          },
        },
      },
    },
  },
};

/**
 * Returns today's UTC date as a Date object at midnight UTC.
 */
function todayUTC(): Date {
  const now = new Date();
  return new Date(Date.UTC(now.getUTCFullYear(), now.getUTCMonth(), now.getUTCDate()));
}

/**
 * Mark a cell as practiced today.
 *
 * The ENTIRE operation runs inside a Prisma interactive transaction per the spec.
 * Transaction algorithm (see design spec "Practice Day Semantics"):
 * 1. Validate ownership chain (inside transaction)
 * 2. Check for existing completion today (including soft-deleted)
 * 3. Branch: 409 / un-soft-delete / create new
 * 4. Calculate raw freshness state and update interval
 * 5. Return updated cell with siblings (for shielding in formatCellResponse)
 *
 * Race handling: If two concurrent INSERTs race, the loser hits P2002
 * (unique constraint violation) — caught by the controller as 409.
 */
export async function completeCell(
  gridId: string,
  rowId: string,
  cellId: string,
  userId: string,
) {
  return prisma.$transaction(async (tx) => {
    // Step 1: Validate ownership chain (inside transaction for ACID)
    const cell = await findOwnedCell(tx, gridId, rowId, cellId, userId);
    if (!cell) return null;

    const today = todayUTC();
    const now = new Date();

    // Step 2: Check for existing completion today (including soft-deleted)
    const existing = await tx.practiceCellCompletion.findFirst({
      where: { practiceCellId: cellId, completionDate: today },
    });

    // Step 3: Branch on result
    if (existing && existing.deletedAt === null) {
      throw new ConflictError('Cell already completed today');
    }

    if (existing && existing.deletedAt !== null) {
      // Un-soft-delete the previously undone completion
      await tx.practiceCellCompletion.update({
        where: { id: existing.id },
        data: { deletedAt: null },
      });
    } else {
      // Create new completion
      await tx.practiceCellCompletion.create({
        data: { practiceCellId: cellId, completionDate: today },
      });
    }

    // Step 4: Calculate raw freshness state and update interval
    const rawState = calculateFreshnessState(today, cell.freshnessIntervalDays, now);
    const stateForInterval = cell.completions.length === 0 ? 'incomplete' : rawState;
    const newInterval = calculateNewInterval(cell.freshnessIntervalDays, stateForInterval);

    await tx.practiceCell.update({
      where: { id: cellId },
      data: { freshnessIntervalDays: newInterval },
    });

    // Step 5: Return updated cell with siblings for formatCellResponse
    return tx.practiceCell.findUniqueOrThrow({
      where: { id: cellId },
      include: CELL_RETURN_INCLUDE,
    });
  });
}
```

```typescript
/**
 * Undo the most recent completion on a cell.
 * Soft-deletes the most recent non-deleted completion.
 * Does NOT restore the interval to its pre-completion value (see design spec decisions).
 *
 * Entire operation inside transaction for ACID. Re-counts completions inside
 * the transaction after the soft-delete to avoid stale count bugs.
 */
export async function undoCompletion(
  gridId: string,
  rowId: string,
  cellId: string,
  userId: string,
) {
  return prisma.$transaction(async (tx) => {
    const cell = await findOwnedCell(tx, gridId, rowId, cellId, userId);
    if (!cell) return null;

    if (cell.completions.length === 0) {
      throw new ConflictError('No completions to undo');
    }

    // Soft-delete the most recent completion (last in asc-ordered array)
    const mostRecent = cell.completions[cell.completions.length - 1];
    await tx.practiceCellCompletion.update({
      where: { id: mostRecent.id },
      data: { deletedAt: new Date() },
    });

    // Re-count non-deleted completions INSIDE the transaction (not from stale pre-fetch)
    const remainingCount = await tx.practiceCellCompletion.count({
      where: { practiceCellId: cellId, deletedAt: null },
    });

    if (remainingCount === 0) {
      // No completions remain — reset interval to 1
      await tx.practiceCell.update({
        where: { id: cellId },
        data: { freshnessIntervalDays: 1 },
      });
    }
    // If completions remain, interval stays as-is (not recoverable — see design decisions)

    return tx.practiceCell.findUniqueOrThrow({
      where: { id: cellId },
      include: CELL_RETURN_INCLUDE,
    });
  });
}

 * Reset a cell's interval to 1. Keeps all completion records.

      throw new ConflictError('No completions to reset');
    }

    await tx.practiceCell.update({
      where: { id: cellId },
      data: { freshnessIntervalDays: 1 },
    });

    return tx.practiceCell.findUniqueOrThrow({
      where: { id: cellId },
      include: CELL_RETURN_INCLUDE,
    });
  });
}
```

```typescript
  completeCell as completeCellModel,
  undoCompletion as undoCompletionModel,

} from '@/models/cell';
import { formatCellResponse } from '@/views/row';

export async function completeCell(gridId: string, rowId: string, cellId: string) {
  try {
    if (!UUID_REGEX.test(gridId) || !UUID_REGEX.test(rowId) || !UUID_REGEX.test(cellId)) {
      return errorResponse('Invalid ID format', 'VALIDATION_ERROR', 400);
    }

    const userId = await getCurrentUserId();
    const cell = await completeCellModel(gridId, rowId, cellId, userId);

    if (!cell) {
      return errorResponse('Cell not found', 'NOT_FOUND', 404);
    }

    return NextResponse.json(formatCellResponse(cell), { status: 201 });
  } catch (error) {
    if (error instanceof AuthenticationError) {
      return errorResponse(error.message, 'AUTHENTICATION_ERROR', 401);
    }
    if (error instanceof ConflictError) {
      return errorResponse(error.message, 'CONFLICT', 409);
    }
    if (error instanceof ValidationError) {
      return errorResponse(error.message, 'VALIDATION_ERROR', 400);
    }
    // P2002 unique constraint violation = concurrent race
    if (
      error instanceof Error &&
      'code' in error &&
      (error as { code: string }).code === 'P2002'
    ) {
      return errorResponse('Cell already completed today', 'CONFLICT', 409);
    }
    return errorResponse('Internal server error', 'INTERNAL_ERROR', 500);
  }
}
```

```typescript
  it('fades highest cell first, then cascades down', async () => {
    // Create a row with 6 cells

    const { grid, row, cells } = result;

    for (const cell of cells) {

    expect(rowData.cells[4].isShielded).toBe(true);
    expect(rowData.cells[0].isShielded).toBe(true);

    // Backdate highest cell's completion to make it decayed
    // interval=1, decayed = >2 days. Set completion to 5 days ago.
    const fiveDaysAgo = new Date();
    fiveDaysAgo.setUTCDate(fiveDaysAgo.getUTCDate() - 5);
    await prisma.practiceCellCompletion.updateMany({

    gridDetail = await getGrid(grid.id);
    gridBody = await gridDetail.json();
    rowData = gridBody.rows[0];
```
